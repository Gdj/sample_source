주요 그리드 옵션

  • datatype: 데이터 타입 ('local', 'json', 'xml')
  • url: 서버 데이터 URL
  • rowNum: 페이지당 행 수
  • rowList: 페이지당 행 수 옵션
  • pager: 페이저 요소 ID
  • caption: 그리드 제목
  • height: 그리드 높이
  • autowidth: 자동 너비 조정
  • multiselect: 다중 선택 가능
  • rownumbers: 행 번호 표시

EVENT

        
        $("#grid").jqGrid({
          // ... 기본 설정
          onSelectRow: function(rowid, status, e) {
            // 행 선택 시 실행
          },
          loadComplete: function(data) {
            // 데이터 로드 완료 시 실행
          },
          beforeRequest: function() {
            // 요청 전 실행
          }
        });

메소드

        
        // 데이터 추가
        $("#grid").jqGrid('addRowData', rowid, data);

        // 행 편집
        $("#grid").jqGrid('editRow', rowid, true);

        // 편집 저장
        $("#grid").jqGrid('saveRow', rowid);

        // 행 삭제
        $("#grid").jqGrid('delRowData', rowid);

        // 선택된 행 ID 가져오기
        var selr = $("#grid").jqGrid('getGridParam', 'selrow');

        // 행 데이터 가져오기
        var data = $("#grid").jqGrid('getRowData', rowid);
